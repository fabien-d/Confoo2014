API w/ jQuery

 var myui = {};

 /**
  * Return first matching element
  * @requires jQuery … */
 myui.qs = function querySelector ( selector ) {
     return $( selector )[ 0 ];
 };

 /**
  * Add class to passed element
  * @requires jQuery … */
 myui.addClass = function addClass ( el, klass ) {
     $( el ).addClass( klass );
 };

code base

 var el = myui.qs( '.conference' );
 myui.addClass( el, 'confoo2014' );

 console.log( el );
 // <div class="conference confoo2014">…</div>

API w/ Native

 var myui = {};

 /**
  * Return first matching element
  * @requires N/A … */
 myui.qs = function querySelector ( selector ) {
     return document.querySelector( selector );
 };

 /**
  * Add class to passed element
  * @requires N/A … */
 myui.addClass = function addClass ( el, klass ) {
     el.classList.add( klass );
 };

code base

 myui.addClass( el, 'confoo2014' );

 console.log( el );

 $.ajax( 'end/point', {

API w/ jQuery

 /**
  * Return jQuery XMLHttpRequest object
  * @requires jQuery … */
 myui.xhr = function xhr ( url, options, success, failure ) {
     return $.ajax( url, {
         type: options.type || 'GET',
         dataType: options.dataType || 'json',
         success: success,
         error: failure
     } );
 };

code base

 var xhr = myui.xhr( 'end/point', null, function () {}, function () {} );

API w/ Native

 /**
  * Return XMLHttpRequest object
  * @requires N/A … */
 myui.xhr = function xhr ( url, options, success, failure ) {
     var request = new XMLHttpRequest();
     request.open( url, options.type || 'GET', true );

     request.onload = function xhrOnLoad () {
         if ( request.status >= 200 && request.status < 400 )
             success( request.responseText );
     };

     request.onerror = function xhrOnError () { failure( request ); };

     request.send();
     return request;
 };
