     var src = '<%= grunt.option( "src" ) %>';
     var tmp = '<%= grunt.option( "tmp" ) %>';

     grunt.config( 'compass', {
         options: {
             sassDir: src + '/sass',
             cssDir: tmp + '/sass'
         },
         dist: {}
     } );

     grunt.loadNpmTasks( 'grunt-contrib-compass' );

1 of 4

build task

 $ > grunt build:dev
 # jshint:     code quality
 # clean:      remove build directory
 # compass:    Sass -> CSS (unminified)
 # versioning: generate CSS/JS config file
 # copy:       copies necessary assets
 # clean:      remove temporary build directories
 $ > grunt build:prod
 # jshint:     code quality
 # clean:      remove build directory
 # compass:    Sass -> CSS (unminified)
 # uglify:     minify and concatenate JS
 # cssmin:     minify CSS
 # versioning: hash min/concat CSS/JS and generate config file
 # copy:       copies necessary assets
 # clean:      remove temporary build directories

4.

debugging

debugging: goal

effectively debug errors

debugging [ compiled, ] concatenated, minified code is hard

debugging in production can be scary

our approach

when a bug is found/reported

1.

build dev environment

  • reproduce the error
  • rule out minification error
  • use devtools on source files
  • update/add unit, regression tests
  • generally all you need

2.

build prod environment

  • reproduce the error
  • use devtools with source maps
  • still have access to source files
  • easier to debug than on the live site
  • update/add unit, regression tests

recap

  • being a front-end developer is hard
  • build tools make you awesome
  • automate the repetitive tasks
  • frictionless project setup
  • reproduce output every time
  • access to dev and production environments locally
  • take the stress out of debugging
